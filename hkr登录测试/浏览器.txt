from selenium import webdriver
import time


class Use_browser:
    #创建统一的浏览器
    def __init__(self, driver):

        self.driver = driver

    #登录网站
    def login(self, username, password):

        self.driver.find_element("id", "loginname").send_keys(username)
        self.driver.find_element("id", "password").send_keys(password)
        time.sleep(1)
        self.driver.find_element("id", "submit").click()
        time.sleep(2)

    #获取登录成功的实际结果
    def getLoginSUccess(self):
        return self.driver.title

    #获取登录失败的实际结果
    def getLogonError(self):
        return self.driver.find_element("id", "msg_uname").text
